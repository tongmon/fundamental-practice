// 정사각 배열 회전
void Rotate(vector<vector<int>>& mat) {
	int N = mat.size();
	for (int x = 0; x < N / 2; x++) {
		for (int y = x; y < N - x - 1; y++) {
			int temp = mat[x][y];
			mat[x][y] = mat[y][N - 1 - x];
			mat[y][N - 1 - x] = mat[N - 1 - x][N - 1 - y];
			mat[N - 1 - x][N - 1 - y] = mat[N - 1 - y][x];
			mat[N - 1 - y][x] = temp;
		}
	}
}

// 최대공약수, 최소공배수
int Gcd(int a, int b) {
	int r = a % b;
	if (r == 0)
		return b;
	return Gcd(b, r);
}

int Lcm(int a, int b) {
	return a * b / Gcd(a, b);
}

// 에라토스테네스의 체 (3000번까지 소수 거르기)
bool Prime[3001] = { 1,1,0, };

void Erathos() {
	for (int i = 4; i < 3001; i += 2)
		Prime[i] = 1;
	for (int i = 3; i <= sqrt(3000); i += 2) {
		if (Prime[i]) continue;
		for (int j = i * i; j < 3001; j += i)
			Prime[j] = 1;
	}
}

// 소수 판정
bool is_Prime(int Num) {
    if (Num <= 1)
        return 0;
    if (!(Num % 2)) {
        if (Num == 2)
            return 1;
        return 0;
    }
    for (int i = 3; i <= sqrt(Num); i+=2) {
        if (!(Num % i))
            return 0;
    }
    return 1;
}
